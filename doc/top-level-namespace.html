post('/logga_in') do
    if session[:username] == session[:info][0][0] and BCrypt::Password.new(session[:info][0][1]) == session[:password]
        redirect('/')
    else
        session[:username] = nil
        session[:password] = nil
        session[:id] = nil
        redirect('/no_access')
    end
end
