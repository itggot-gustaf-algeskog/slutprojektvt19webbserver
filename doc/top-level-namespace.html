post('/sign_in') do
    info = sign_in(params)
    session[:username] = params["username"]
    session[:password] = params["password"]

    if info == []
        session[:username] = nil
        session[:password] = nil
        session[:id] = nil
        redirect('/no_access')
    else
        if session[:username] == info[0][0] and BCrypt::Password.new(info[0][1]) == session[:password]
            session[:id] = info[0][2]
            redirect('/')
        else
            session[:username] = nil
            session[:password] = nil
            session[:id] = nil
            redirect('/no_access')
        end
    end
end
